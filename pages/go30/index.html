type UserAttrFunc func(*User) //设置User属性的 函数类型
type UserAttrFuncs []UserAttrFunc //User属性的 函数类型切片

// 添加方法，遍历切片，对函数赋值
func(this UserAttrFuncs) apply(u *User)  {
    //遍历函数切片 
   for _,f:=range this{
       //每个赋值函数,成User类型的结构体
       /* 
       func(u *User) {
      u.Id=id
     */ 
      f(u)
   }
}


// 返回一个赋值的函数
func WithUserID(id int) UserAttrFunc  {
   return func(u *User) {
      u.Id=id
   }
}
func WithUserName(name string) UserAttrFunc  {
   return func(u *User) {
      u.Name=name
   }
}

# 3.3、map的单个排序

package Map

import (
	"fmt"
	"sort"
)

type User map[string]interface{}

func NewUser() User {
	return make(map[string]interface{})
}
func (this User) With(k string, v interface{}) User {
	this[k] = v
	return this
}

//定义一个排序方法
func (this User) Fields() []string {
    // 遍历出k
	keys := []string{}
	for k, _ := range this {
		keys = append(keys, k)
	}
	// 正排
	//sort.Strings(sort.StringSlice(keys))
    // 反排
	sort.Sort(sort.Reverse(sort.StringSlice(keys)))
	return keys
}

func (this User) String() string {
	str := ""
    // 修改 打印方法，默认执行排序
	for index, k := range this.Fields() {
		fmt.Printf("%d,%v->%v\n", index+1, k, this[k])
	}
	return str
}

package Utils

import (
	"bytes"
	"strings"
)

func Join(str ...string) string {
	ret := ""
	for _, s := range str {
		ret += s
	}
	return ret
}

func Join2(str ...string) string {
	return strings.Join(str, "")
}
func Join3(str ...string) string {
	var bf bytes.Buffer
	for _, s := range str {
		bf.WriteString(s)
	}
	return bf.String()
}

# 6.3、map映射成struct

package main

import (
	"fmt"
	"reflect"
)
// 需要映射的结构体
type User struct {
	UserId   int
	UserName string
}
// 主体函数
func Map2Struct(m map[string]interface{}, u interface{}) {
    // 对结构体的初始化
	v := reflect.ValueOf(u)
    // 类型判断必须为指针
	if v.Kind() == reflect.Ptr {
        //指针对应的函数
		v = v.Elem()
        //不是struct报错
		if v.Kind() != reflect.Struct {
			panic("must struct")

		}
         //匿名函数 寻找KEY的值，返回空接口
		findFromMap := func(key string) interface{} {
			for k, v := range m {
				if k == key {
					return v
				}
			}
			return nil
		}

		for i := 0; i < v.NumField(); i++ {
            // 获取 值
			getValue := findFromMap(v.Type().Field(i).Name)
            // 判断不等于空并且值类型相等于的时候 对u进行赋值
			if getValue != nil && reflect.ValueOf(getValue).Kind() == v.Field(i).Kind() {
				v.Field(i).Set(reflect.ValueOf(getValue))
			}
		}
	} else {
		panic("must ptr")
	}
}
func main() {
	u := &User{}
	m := map[string]interface{}{
		"id":       123,
		"UserId":   123,
		"UserName": "xiongmao",
		"age":      19,
	}
	Map2Struct(m, u)
	fmt.Println(u)
}

# 6.4、map映射成struct的tag

package main

import (
	"fmt"
	"reflect"
)

type User struct {
	UserId   int    `name:"uid"`
	UserName string `name:"uname"`
}

func Map2Struct(m map[string]interface{}, u interface{}) {
	v := reflect.ValueOf(u)
	if v.Kind() == reflect.Ptr {
		v = v.Elem()
		if v.Kind() != reflect.Struct {
			panic("must struct")

		}
		// 将tag传入 添加匹配Tag
		findFromMap := func(key string, nameTag string) interface{} {
			for k, v := range m {
                // 匹配到tag时候返回值
				if k == key || k == nameTag {
					return v
				}
			}
			return nil
		}

		for i := 0; i < v.NumField(); i++ {
            //传入tag
			getValue := findFromMap(v.Type().Field(i).Name, v.Type().Field(i).Tag.Get("name"))

			if getValue != nil && reflect.ValueOf(getValue).Kind() == v.Field(i).Kind() {
				v.Field(i).Set(reflect.ValueOf(getValue))
				fmt.Println(v.Field(i).Interface())
			}
		}
	} else {
		panic("must ptr")
	}
}
func main() {
	u := &User{}
	m := map[string]interface{}{
		"id":       123,
		"uid":      123,
		"uname": "xiongmao",
		"age":      19,
	}
	Map2Struct(m, u)
	fmt.Println(u)
}
