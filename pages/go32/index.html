# 3、强化管道符得可变参数

package main

import (
	"fmt"
)

type Cmd func(list []int) chan int
type PipCmd func(int chan int) chan int

func Evens(list []int) chan int {
	c := make(chan int)
	go func() {
		defer close(c)
		for _, num := range list {
			if num%2 == 0 {
				c <- num
			}
		}
	}()

	return c
}

func Multiply(in chan int) chan int {
	out := make(chan int)
	go func() {
		defer close(out)
		for num := range in {
			out <- num * 10
		}
	}()

	return out
}
func p(args []int, c1 Cmd, c2 ...PipCmd) chan int {
	ret := c1(args)
	if len(c2) == 0 {
		return ret
	}
	retlist := make([]chan int, 0)
	// 遍历一共有几条命令
	for index, c := range c2 {
		// 第一次 将结果存入retlist
		if index == 0 {
			retlist = append(retlist, c(ret))
			// 否则 存入	retlist的最后一个
		} else {
			getChan := retlist[len(retlist)-1]
			retlist = append(retlist, c(getChan))
		}
	}
	return retlist[len(retlist)-1]
}

func main() {
	nums := []int{2, 3, 4, 5, 7, 8, 64, 9, 4}
	ret := p(nums, Evens, Multiply, Multiply)
	for num := range ret {
		fmt.Printf("%d ", num)
	}

}

# 4、管道模式的多路复用

package main

import (
	"fmt"
	"sync"
	"time"
)

type Cmd func(list []int) chan int
type PipCmd func(int chan int) chan int

func Evens(list []int) chan int {
	c := make(chan int)
	go func() {
		defer close(c)
		for _, num := range list {
			if num%2 == 0 {
				c <- num
			}
		}
	}()

	return c
}

func Multiply(in chan int) chan int {
	out := make(chan int)
	go func() {
		defer close(out)
		for num := range in {
			time.Sleep(time.Millisecond * 300)
			out <- num * 10
		}
	}()

	return out
}
func p(args []int, c1 Cmd, c2 ...PipCmd) chan int {
	ret := c1(args)
	if len(c2) == 0 {
		return ret
	}
	retlist := make([]chan int, 0)
	// 遍历一共有几条命令
	for index, c := range c2 {
		// 第一次 将结果存入retlist
		if index == 0 {
			retlist = append(retlist, c(ret))
			// 否则 存入	retlist的最后一个
		} else {
			getChan := retlist[len(retlist)-1]
			retlist = append(retlist, c(getChan))
		}
	}
	return retlist[len(retlist)-1]
}

func p2(args []int, c1 Cmd, c2 ...PipCmd) chan int {
	ret := c1(args)
	out := make(chan int)
	wg := sync.WaitGroup{}
	//  遍历命令
	for _, c := range c2 {
		// 执行cmd
		getChan := c(ret)
		wg.Add(1)
		// 开启协程有值返回
		go func(in chan int) {
			defer wg.Done()
			for v := range in {
				out <- v
			}
		}(getChan)
	}
	// 协程wg.Wait()不等待直接返回
	go func() {
		defer close(out)
		wg.Wait()
	}()
	return out

}

func main() {
	nums := []int{2, 3, 4, 5, 7, 8, 64, 9, 4}
	ret := p2(nums, Evens, Multiply, Multiply)
	for num := range ret {
		fmt.Printf("%d ", num)
	}

}

package getdata

import (
	"encoding/csv"
	"fmt"
	"log"
	"os"
	"strconv"
	"string/src/pipeline/AppInit"
)

type Book struct {
	BookId   int    `gorm:"column:book_id"`
	BookName string `gorm:"column:book_name"`
}
type BookList struct {
	Data []*Book
	Page int
}

const sql = "select * from books order by book_id limit ? offset ? "

func ReadData() {
	page := 1
	pagesize := 1000
	for {
        // 初始化结构体
		booklist := &BookList{make([]*Book, 0), page}
        // 取出1千行
		db := AppInit.GetDB().Raw(sql, pagesize, (page-1)*pagesize).Find(&booklist.Data)
        // 直到最后一行
		if db.Error != nil || db.RowsAffected == 0 {
			break
		}
        // 保存
		err := SaveData(booklist)
		if err != nil {
			log.Println(err)
		}
		page++
	}
}

//写入到csv文件
func SaveData(data *BookList) error {
	file := fmt.Sprintf("./src/pipeline/csv/%d.csv", data.Page)
	csvFile, err := os.OpenFile(file, os.O_RDWR|os.O_CREATE|os.O_TRUNC, 0666)
	if err != nil {
		return err
	}
	defer csvFile.Close()
	w := csv.NewWriter(csvFile) //创建一个新的写入文件流
	header := []string{"book_id", "book_name"}
	export := [][]string{
		header,
	}
	for _, d := range data.Data {
		cnt := []string{
			strconv.Itoa(d.BookId),
			d.BookName,
		}
		export = append(export, cnt)
	}
	err = w.WriteAll(export)
	if err != nil {
		return err
	}
	w.Flush()
	return nil
}

# 6、mysql导出csv的应用的改造

package getdata

import (
	"encoding/csv"
	"fmt"
	"os"
	"strconv"
	"string/src/pipeline/AppInit"
	"sync"
	"time"
)

type Book struct {
	BookId   int    `gorm:"column:book_id"`
	BookName string `gorm:"column:book_name"`
}
type BookList struct {
	Data []*Book
	Page int
}
type Result struct {
	Page int
	Err  error
}
type inChan chan *BookList
type OutChan chan *Result

type DataCmd func() inChan
type DataPipeCmd func(in inChan) OutChan

const sql = "select * from books order by book_id limit ? offset ? "

func ReadData() inChan {
	page := 1
	pagesize := 1000
	in := make(inChan)
	go func() {
		defer close(in)
		for {
			booklist := &BookList{make([]*Book, 0), page}
			db := AppInit.GetDB().Raw(sql, pagesize, (page-1)*pagesize).Find(&booklist.Data)
			if db.Error != nil || db.RowsAffected == 0 {
				break
			}
			in <- booklist
			page++
		}
	}()
	return in

}

func WriteData(in inChan) OutChan {
	out := make(OutChan)
	go func() {
		defer close(out)
		for d := range in {
			out <- &Result{Page: d.Page, Err: SaveData(d)}
		}
	}()
	return out
}
func Pipe(c1 DataCmd, cs ...DataPipeCmd) OutChan {
	in := c1()
	out := make(OutChan)
	wg := sync.WaitGroup{}
	for _, c := range cs {
		getChan := c(in)
		wg.Add(1)
		go func(input OutChan) {
			defer wg.Done()
			for v := range input {
				out <- v
			}
		}(getChan)

	}
	go func() {
		wg.Wait()
		defer close(out)
	}()
	return out
}


//写入到csv文件
func SaveData(data *BookList) error {
	time.Sleep(time.Millisecond * 500)
	file := fmt.Sprintf("./src/pipeline/csv/%d.csv", data.Page)
	csvFile, err := os.OpenFile(file, os.O_RDWR|os.O_CREATE|os.O_TRUNC, 0666)
	if err != nil {
		return err
	}
	defer csvFile.Close()
	w := csv.NewWriter(csvFile) //创建一个新的写入文件流
	header := []string{"book_id", "book_name"}
	export := [][]string{
		header,
	}
	for _, d := range data.Data {
		cnt := []string{
			strconv.Itoa(d.BookId),
			d.BookName,
		}
		export = append(export, cnt)
	}
	err = w.WriteAll(export)
	if err != nil {
		return err
	}
	w.Flush()
	return nil
}
func Test() {
	out := Pipe(ReadData, WriteData)
	for o := range out {
		fmt.Printf("%d,csv文件执行完成结果：%v\n", o.Page, o.Err)
	}
}

package getdata

import (
	"fmt"
	"string/src/pipeline/AppInit"
	"time"

	"log"
)

type Book struct {
	BookId   int    `gorm:"column:book_id"`
	BookName string `gorm:"column:book_name"`
}

func (this *Book) String() string {
	return fmt.Sprintf("bookid:%d,book_name:%s\n", this.BookId, this.BookName)
}

type BookList struct {
	Data []*Book
	Page int
}
type Result struct {
	Page int
	Err  error
}

const sql = "select * from books order by book_id limit ? offset ?"
//获取页面数据
func GetPage(args ...interface{}) InChan {
	in := make(InChan)
	go func() {
		defer close(in)
		for i := 1; i <= 10; i++ {
			in <- i
		}
	}()
	return in

}
//获取数据
func GetData(in InChan) OutChan {
	out := make(OutChan)
	go func() {
		defer close(out)
		for d := range in {
			page := d.(int)
			pagesize := 1000
			booklist := &BookList{make([]*Book, 0), page}
			db := AppInit.GetDB().Raw(sql, pagesize, (page-1)*pagesize).Find(&booklist.Data)
			if db.Error != nil {
				log.Println(db.Error)
			}
			out <- booklist.Data
		}
	}()
	return out
}

//新增模拟处理数据
func DoData(in InChan) OutChan {
	out := make(OutChan)
	go func() {
		defer close(out)
		for d := range in {
			v := d.([]*Book)
			time.Sleep(time.Second * 1)
			out <- fmt.Sprintf("处理了%d条数据,%d\n", len(v), time.Now().Unix())
		}
	}()
	return out
}

func PipeTest() {
    // 初始化结构体
	p1 := NewPipe()
    // 结构体的第一次函数
	p1.SetCmd(GetPage)
    // 结构体的管道函数
	p1.SetPipeCmd(GetData, 5)
    // 获取值
	out := p1.Exec()
    // 初始化新的结构体
	p2 := NewPipe()
    // CmdFunc 不需要所以传入空的func
	p2.SetCmd(func(args ...interface{}) InChan {
		return InChan(out)
	})
    // 设置管道函数
	p2.SetPipeCmd(DoData, 2)
    // 获取值
	out2 := p2.Exec()

	for item := range out2 {
		fmt.Println(item)
	}

}

package getdata

import "sync"
// 定义chan 空接口
type InChan chan interface{}
type OutChan chan interface{}
//定义结构体普通命令和多路复用
type CmdFunc func(args ...interface{}) InChan
type PipeCmdFunc func(in InChan) OutChan

//定义结构体管道工具
type Pipe struct {
	Cmd     CmdFunc
	PipeCmd PipeCmdFunc
	Count   int
}
//初始化pip默认为1
func NewPipe() *Pipe {
	return &Pipe{Count: 1}
}
//设置结构体的cmd
func (this *Pipe) SetCmd(c CmdFunc) {
	this.Cmd = c
}
//设置结构体的pipecmd
func (this *Pipe) SetPipeCmd(c PipeCmdFunc, count int) {
	this.PipeCmd = c
	this.Count = count
}
//结构体的方法
func (this *Pipe) Exec(args ...interface{}) OutChan {
    //获取数据
	in := this.Cmd(args)
    //初始化一个chan 空接口
	out := make(OutChan)
    //协程执行返回chan
	wg := sync.WaitGroup{}
	for i := 0; i < this.Count; i++ {
		getChan := this.PipeCmd(in)
		wg.Add(1)
		go func(input OutChan) {
			defer wg.Done()
			for v := range input {
				out <- v
			}
		}(getChan)
	}
	go func() {
		defer close(out)
		wg.Wait()
	}()
	return out
}
